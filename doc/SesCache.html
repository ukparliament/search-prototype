def exists_in_cache?(ses_id)
  # puts "Looking for #{cache_key(ses_id)} in cache..." if Rails.env.development?
  if cache_store.exist?(cache_key(ses_id))
    # puts "...found"
    true
  else
    # puts "... not found"
    false
  end
end
